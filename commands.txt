git config --global user.name "username" [create your username]

git config --global user.email "email" [create your email]

git init            [initialize the git empty commands]

git status          [to check the untracked files]

git add -A          [add all untracked files to staging area]

git commit -m "commit message"     [to add the staging files to local repository]

git branch      [to list the all branch]

git remote -v       [to list the all remote]

git remote add remote_name "remote_path   [to add remote repository to local]

git push remote_name branch_name  [push]
